Script 1

This script makes you double jump


local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")
local userInputService = game:GetService("UserInputService")

local jumpCount = 0
local maxJumps = 2

character:WaitForChild("HumanoidRootPart")
humanoid:GetPropertyChangedSignal("FloorMaterial"):Connect(function()

    if humanoid.FloorMaterial ~= Enum.Material.Air then
        jumpCount = 0
    end
end)

userInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end

    if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode == Enum.KeyCode.Space then
        if jumpCount < maxJumps then
            humanoid:ChangeState(Enum.HumanoidStateType.Physics)
            humanoid:Move(Vector3.new(0, 50, 0))
            jumpCount = jumpCount + 1
        end
    end
end)

Script 4

This script adds a health regen function


Health Regen Script
local regenAmount = 5
ocal regenInterval = 5
                                
game.Players.PlayerAdded:Connect(function(player)

player.CharacterAdded:Connect(function(character)

local humanoid = character:WaitForChild("Humanoid")
                                        
while humanoid and humanoid.Health > 0 do

wait(regenInterval)
                                            
if humanoid.Health < humanoid.MaxHealth then
humanoid.Health = humanoid.Health + regenAmount
                                                
Ensure health doesn't exceed max health
if humanoid.Health > humanoid.MaxHealth then
humanoid.Health = humanoid.MaxHealth
end
end
end
end)
end)

Script 6

This script enables you to set a spawn location


ServerScriptService - script
game.Players.PlayerAdded:Connect(function(player)
    -- Wait until the player's character is added
    player.CharacterAdded:Connect(function(character)
        -- Get the spawn location (the part you created)
        local spawnLocation = game.Workspace:WaitForChild("SpawnLocation")

        -- Set the character's position to the spawn location
        character:SetPrimaryPartCFrame(spawnLocation.CFrame)
    end)
end)

Workspace
game.Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(character)
        -- Create a list of spawn locations
        local spawnLocations = {}
        for _, spawnPart in pairs(game.Workspace:GetChildren()) do
            if spawnPart.Name:match("SpawnLocation") then
                table.insert(spawnLocations, spawnPart)
            end
        end

        -- Randomly choose a spawn location
        local randomSpawn = spawnLocations[math.random(1, #spawnLocations)]

        -- Set the player's spawn location
        character:SetPrimaryPartCFrame(randomSpawn.CFrame)
    end)
end)
